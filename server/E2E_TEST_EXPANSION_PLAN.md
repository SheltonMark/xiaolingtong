```typescript
test('should sync data after going offline and back online', async ({ page, context }) => {
  // 模拟离线
  await context.setOffline(true);

  // 在离线状态下修改数据
  await page.fill('[data-testid="balance-input"]', '100');
  await page.click('[data-testid="save-btn"]');

  // 验证本地缓存
  const cachedValue = await page.locator('[data-testid="balance"]').textContent();
  expect(cachedValue).toBe('100');

  // 恢复连接
  await context.setOffline(false);
  await page.waitForTimeout(1000);

  // 验证与服务器同步
  const syncedValue = await page.locator('[data-testid="balance"]').textContent();
  expect(syncedValue).toBe('100');
});
```

**示例**: 不同屏幕尺寸的布局测试
```typescript
test.describe('Responsive Design', () => {
  const viewports = [
    { name: 'mobile', width: 375, height: 667 },
    { name: 'tablet', width: 768, height: 1024 },
    { name: 'desktop', width: 1920, height: 1080 }
  ];

  viewports.forEach(viewport => {
    test(`should display correctly on ${viewport.name}`, async ({ page }) => {
      await page.setViewportSize(viewport);
      await page.goto('/pages/mine/mine');

      // 验证布局
      const card = page.locator('[data-testid="app-card"]');
      const box = await card.boundingBox();

      expect(box.width).toBeLessThanOrEqual(viewport.width - 32);
    });
  });
});
```
